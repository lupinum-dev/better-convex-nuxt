## How It Works

1. **User action**: User clicks button/submits form
2. **Optimistic update**: UI updates immediately with expected result
3. **Server request**: Mutation sent to Convex
4. **Resolution**:
   - Success: Server data replaces optimistic data
   - Failure: Optimistic changes rolled back automatically

---

## Basic Example

```ts
const createTodo = useConvexMutation(api.todos.create, {
  optimisticUpdate: (ctx, args) => {
    // Update local query cache immediately
    ctx.query(api.todos.list, {}).update((current) => {
      const optimisticTodo = {
        _id: crypto.randomUUID() as Id<'todos'>,
        _creationTime: Date.now(),
        text: args.text,
        completed: false,
      }
      return current ? [optimisticTodo, ...current] : [optimisticTodo]
    })
  },
})
```

## Fluent Builder API

Optimistic updates use a fluent builder API accessed through the `ctx` parameter.

### Regular Queries

#### `ctx.query(query, args).update(updater)`

Update a query result with an updater function.

```ts
optimisticUpdate: (ctx, args) => {
  ctx.query(api.todos.list, {}).update((current) => {
    const newTodo = { _id: crypto.randomUUID(), text: args.text }
    return current ? [newTodo, ...current] : [newTodo]
  })
}
```

#### `ctx.query(query, args).set(value)`

Set query data directly without transformation.

```ts
optimisticUpdate: (ctx, args) => {
  ctx.query(api.users.get, { userId: args.userId }).set({
    ...currentUser,
    name: args.newName,
  })
}
```

### Paginated Queries

#### `ctx.paginatedQuery(query, args).insertAtTop(item)`

Insert item at the top of paginated results. Use for newest-first lists.

```ts
const sendMessage = useConvexMutation(api.messages.send, {
  optimisticUpdate: (ctx, args) => {
    ctx.paginatedQuery(api.messages.list, {}).insertAtTop({
      _id: crypto.randomUUID() as Id<'messages'>,
      _creationTime: Date.now(),
      body: args.body,
      authorId: currentUser._id,
    })
  },
})
```

#### `ctx.paginatedQuery(query, args).insertAtPosition(item, position)`

Insert item at a specific position.

```ts
optimisticUpdate: (ctx, args) => {
  ctx.paginatedQuery(api.tasks.byPriority, {}).insertAtPosition(
    {
      _id: crypto.randomUUID() as Id<'tasks'>,
      _creationTime: Date.now(),
      title: args.title,
      priority: args.priority,
    },
    2, // position index
  )
}
```

#### `ctx.paginatedQuery(query, args).insertAtBottomIfLoaded(item)`

Insert at bottom only if all pages are loaded.

```ts
optimisticUpdate: (ctx, args) => {
  ctx.paginatedQuery(api.messages.listOldestFirst, {}).insertAtBottomIfLoaded({
    _id: crypto.randomUUID() as Id<'messages'>,
    _creationTime: Date.now(),
    body: args.body,
  })
}
```

#### `ctx.paginatedQuery(query, args).updateItem(id, updater)`

Update a specific item in paginated results by ID.

```ts
optimisticUpdate: (ctx, args) => {
  ctx.paginatedQuery(api.tasks.list, {}).updateItem(args.taskId, (task) => ({
    ...task,
    completed: !task.completed,
  }))
}
```

#### `ctx.paginatedQuery(query, args).deleteItem(id)`

Remove an item from paginated results by ID.

```ts
optimisticUpdate: (ctx, args) => {
  ctx.paginatedQuery(api.messages.list, {}).deleteItem(args.messageId)
}
```

### Escape Hatch

#### `ctx.store`

Access the raw `OptimisticLocalStore` for advanced use cases not covered by the fluent API.

```ts
optimisticUpdate: (ctx, args) => {
  const rawStore = ctx.store
  // Use raw Convex optimistic local store methods directly
}
```

## Complete Examples

### Todo App

```vue
<script setup lang="ts">
import { api } from '~~/convex/_generated/api'
import type { Id } from '~~/convex/_generated/dataModel'

const { data: todos } = await useConvexQuery(api.todos.list, {})

// Add todo with optimistic update
const addTodo = useConvexMutation(api.todos.create, {
  optimisticUpdate: (ctx, args) => {
    ctx.query(api.todos.list, {}).update((current) => {
      const optimistic = {
        _id: crypto.randomUUID() as Id<'todos'>,
        _creationTime: Date.now(),
        text: args.text,
        completed: false,
      }
      return current ? [optimistic, ...current] : [optimistic]
    })
  },
})

// Toggle todo with optimistic update
const toggleTodo = useConvexMutation(api.todos.toggle, {
  optimisticUpdate: (ctx, args) => {
    ctx.query(api.todos.list, {}).update((current) => {
      if (!current) return current
      return current.map((todo) =>
        todo._id === args.id ? { ...todo, completed: !todo.completed } : todo,
      )
    })
  },
})

// Delete todo with optimistic update
const deleteTodo = useConvexMutation(api.todos.remove, {
  optimisticUpdate: (ctx, args) => {
    ctx.query(api.todos.list, {}).update((current) => {
      if (!current) return current
      return current.filter((todo) => todo._id !== args.id)
    })
  },
})

const newTodoText = ref('')

async function handleAdd() {
  if (!newTodoText.value.trim()) return
  await addTodo({ text: newTodoText.value })
  newTodoText.value = ''
}
</script>

<template>
  <div class="todo-app">
    <form @submit.prevent="handleAdd">
      <input v-model="newTodoText" :disabled="addTodo.pending.value" />
      <button :disabled="addTodo.pending.value">Add</button>
    </form>

    <ul>
      <li v-for="todo in todos" :key="todo._id">
        <input type="checkbox" :checked="todo.completed" @change="toggleTodo({ id: todo._id })" />
        <span :class="{ completed: todo.completed }">{{ todo.text }}</span>
        <button @click="deleteTodo({ id: todo._id })">Delete</button>
      </li>
    </ul>
  </div>
</template>
```

### Chat with Paginated Messages

```vue
<script setup lang="ts">
import { api } from '~~/convex/_generated/api'
import type { Id } from '~~/convex/_generated/dataModel'

const { user } = useConvexAuth()

const {
  results: messages,
  status,
  loadMore,
} = await useConvexPaginatedQuery(
  api.messages.list,
  { channelId: props.channelId },
  { initialNumItems: 50 },
)

// Send message with optimistic update
const sendMessage = useConvexMutation(api.messages.send, {
  optimisticUpdate: (ctx, args) => {
    ctx.paginatedQuery(api.messages.list, { channelId: args.channelId }).insertAtTop({
      _id: crypto.randomUUID() as Id<'messages'>,
      _creationTime: Date.now(),
      body: args.body,
      channelId: args.channelId,
      authorId: user.value?.id as Id<'users'>,
      authorName: user.value?.name ?? 'You',
    })
  },
})

// Delete message with optimistic update
const deleteMessage = useConvexMutation(api.messages.delete, {
  optimisticUpdate: (ctx, args) => {
    ctx.paginatedQuery(api.messages.list, { channelId: props.channelId }).deleteItem(args.messageId)
  },
})

const messageText = ref('')

async function handleSend() {
  if (!messageText.value.trim()) return
  await sendMessage({
    channelId: props.channelId,
    body: messageText.value,
  })
  messageText.value = ''
}
</script>

<template>
  <div class="chat">
    <div class="messages">
      <div v-for="msg in messages" :key="msg._id" class="message">
        <strong>{{ msg.authorName }}</strong>
        <p>{{ msg.body }}</p>
        <button v-if="msg.authorId === user?.id" @click="deleteMessage({ messageId: msg._id })">
          Delete
        </button>
      </div>
      <button v-if="status === 'ready'" @click="loadMore(50)">Load older messages</button>
    </div>

    <form @submit.prevent="handleSend" class="compose">
      <input v-model="messageText" :disabled="sendMessage.pending.value" />
      <button :disabled="sendMessage.pending.value">Send</button>
    </form>
  </div>
</template>
```

## Best Practices

### 1. Generate temporary IDs

Always use unique temporary IDs for optimistic items:

```ts
const optimisticId = crypto.randomUUID() as Id<'todos'>
```

### 2. Include all required fields

Optimistic items should match the query return shape:

```ts
// If your query returns these fields...
{
  _id: Id<'todos'>,
  _creationTime: number,
  text: string,
  completed: boolean,
  userId: Id<'users'>,
}

// ...include them all in optimistic update
ctx.query(api.todos.list, {}).update((current) => [
  {
    _id: crypto.randomUUID() as Id<'todos'>,
    _creationTime: Date.now(),
    text: args.text,
    completed: false,
    userId: currentUserId,  // Include all fields!
  },
  ...(current ?? []),
])
```

### 3. Handle missing data gracefully

```ts
updater: (current) => {
  if (!current) return current  // Don't crash if query not loaded
  return current.map(item => /* transform */)
}
```

### 4. Match query args exactly

```ts
// If your query uses these args...
const { data } = await useConvexQuery(api.todos.list, { status: 'active' })

// ...match them in optimistic update
ctx.query(api.todos.list, { status: 'active' }).update((current) => /* ... */)
```

## Common Mistakes

### 1. Forgetting to handle rollback

```ts
// WRONG: Assuming mutation will succeed
const createTodo = useConvexMutation(api.todos.create, {
  optimisticUpdate: (ctx, args) => {
    // Update happens...
  },
})

async function handleCreate() {
  await createTodo({ text }) // If this fails, UI auto-rollbacks
  toast.success('Created!') // But toast shows anyway!
}

// RIGHT: Handle errors properly
async function handleCreate() {
  try {
    await createTodo({ text })
    toast.success('Created!')
  } catch {
    toast.error('Failed to create')
  }
}
```

### 2. Modifying current value directly

```ts
// WRONG: Mutating the array
updater: (current) => {
  current?.push(newItem) // Don't mutate!
  return current
}

// RIGHT: Return new array
updater: (current) => {
  return current ? [...current, newItem] : [newItem]
}
```

### 3. Not matching paginated query shape

```ts
// WRONG: Missing required pagination fields
ctx.paginatedQuery(api.messages.list, {}).insertAtTop({
  body: args.body, // Missing _id, _creationTime, etc.
})

// RIGHT: Include all fields
ctx.paginatedQuery(api.messages.list, {}).insertAtTop({
  _id: crypto.randomUUID() as Id<'messages'>,
  _creationTime: Date.now(),
  body: args.body,
  authorId: currentUser._id,
})
```
