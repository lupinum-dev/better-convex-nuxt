## Conditional Query Based on Route Params

Skip the query until the route param is available. Pass `null` args to skip execution.

```vue
<script setup lang="ts">
import { api } from '#trellis/api'

const route = useRoute()

const { data: post } = await useConvexQuery(api.posts.get, () => {
  const id = route.params.id as string | undefined
  return id ? { id } : null
})
</script>

<template>
  <article v-if="post">
    <h1>{{ post.title }}</h1>
    <p>{{ post.content }}</p>
  </article>
</template>
```

## Optimistic Delete from a List

Remove the item from the UI immediately. If the server rejects the delete, it rolls back.

```vue
<script setup lang="ts">
import { api } from '#trellis/api'

const { data: tasks } = await useConvexQuery(api.tasks.list, {})

const removeTask = useConvexMutation(api.tasks.remove, {
  optimisticUpdate: (ctx, args) => {
    ctx
      .query(api.tasks.list, {})
      .update((current) => (current ? current.filter((t) => t._id !== args.id) : []))
  },
})
</script>

<template>
  <ul>
    <li v-for="task in tasks" :key="task._id">
      {{ task.text }}
      <button @click="removeTask({ id: task._id })">Delete</button>
    </li>
  </ul>
</template>
```

## File Upload with Preview

Show a local preview before the upload finishes. Use the File's object URL for instant feedback.

```vue
<script setup lang="ts">
import { api } from '#trellis/api'

const upload = useConvexUpload(api.files.generateUploadUrl)
const previewUrl = ref<string | null>(null)

async function handleFile(event: Event) {
  const file = (event.target as HTMLInputElement).files?.[0]
  if (!file) return

  previewUrl.value = URL.createObjectURL(file)
  const storageId = await upload(file)
  // Save storageId to your document
}
</script>

<template>
  <input type="file" accept="image/*" @change="handleFile" />
  <img v-if="previewUrl" :src="previewUrl" alt="Preview" class="max-w-xs" />
  <p v-if="upload.pending.value">Uploading: {{ upload.progress.value }}%</p>
</template>
```

## Role-Gated Button Visibility

Show or hide actions based on the user's permissions. The `can()` function returns a `ComputedRef<boolean>`.

```vue
<script setup lang="ts">
import { api } from '#trellis/api'

const { can } = usePermissions()
const { data: post } = await useConvexQuery(api.posts.get, { id: props.id })

const publish = useConvexMutation(api.posts.publish)
const remove = useConvexMutation(api.posts.remove)
</script>

<template>
  <article v-if="post">
    <h1>{{ post.title }}</h1>

    <div class="actions">
      <button v-if="can('post.publish')" @click="publish({ id: post._id })">Publish</button>
      <button v-if="can('post.delete')" @click="remove({ id: post._id })">Delete</button>
    </div>
  </article>
</template>
```

## Server-Side Webhook That Calls a Convex Mutation

Use `serverConvexMutation` in a Nitro route to handle incoming webhooks. The `authMode: 'trusted'` flag uses the trusted caller path instead of a user session.

```ts [server/api/webhooks/stripe.post.ts]
export default defineEventHandler(async (event) => {
  const body = await readBody(event)

  await serverConvexMutation(
    api.billing.processWebhook,
    {
      eventId: body.id,
      type: body.type,
      data: body.data.object,
    },
    { authMode: 'trusted' },
  )

  return { received: true }
})
```

## List-to-Detail with Cached Data

Avoid the loading flash when navigating from a list to a detail page. Use `useCachedQuery` to seed the detail query from the list data.

```vue
<script setup lang="ts">
import { api } from '#trellis/api'

const props = defineProps<{ id: string }>()

const { data: post, isFromCache } = await useCachedQuery(api.posts.get, () => ({ id: props.id }), {
  from: {
    query: api.posts.list,
    args: {},
    find: (posts) => posts.find((p) => p._id === props.id),
  },
})
</script>

<template>
  <article v-if="post">
    <span v-if="isFromCache" class="text-xs text-gray-400">Loading fresh data...</span>
    <h1>{{ post.title }}</h1>
    <p>{{ post.content }}</p>
  </article>
</template>
```

## Offline-Aware UI Banner

Show a reconnection banner when the WebSocket drops. The connection state updates automatically when the client reconnects.

```vue
<script setup lang="ts">
const { isConnected, isReconnecting } = useConvexConnectionState()
</script>

<template>
  <div v-if="isReconnecting" class="fixed top-0 inset-x-0 bg-yellow-500 text-center py-1 text-sm">
    Reconnecting...
  </div>
  <slot />
</template>
```

## Global Error Reporting with Hooks

Catch all mutation and action errors in one place using runtime hooks. Useful for Sentry, LogRocket, or similar error tracking.

```ts [plugins/error-tracking.ts]
export default defineNuxtPlugin((nuxtApp) => {
  nuxtApp.hook('trellis:mutation:error', ({ functionPath, error, duration }) => {
    reportError({
      type: 'convex-mutation',
      function: functionPath,
      message: error.message,
      duration,
    })
  })

  nuxtApp.hook('trellis:action:error', ({ functionPath, error, duration }) => {
    reportError({
      type: 'convex-action',
      function: functionPath,
      message: error.message,
      duration,
    })
  })
})
```
