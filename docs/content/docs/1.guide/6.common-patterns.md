## SSR Query In A Page

```vue
<script setup lang="ts">
import { api } from '#trellis/api'

const {
  data: tasks,
  status,
  error,
} = await useConvexQuery(api.tasks.list, {
  status: 'active',
})
</script>
```

## Query With Default And Transform

```vue
<script setup lang="ts">
import { api } from '#trellis/api'

const selectedTag = ref<string | null>(null)

const { data: labels } = await useConvexQuery(
  api.tags.list,
  () => (selectedTag.value ? { tag: selectedTag.value } : null),
  {
    default: () => [],
    transform: (items) => items.map((item) => item.label),
    keepPreviousData: true,
  },
)
</script>
```

## Mutation With Optimistic Update

```vue
<script setup lang="ts">
import { api } from '#trellis/api'

const addTask = useConvexMutation(api.tasks.create, {
  optimisticUpdate: (ctx, args) => {
    prependTo(
      ctx,
      api.tasks.list,
      {},
      {
        _id: 'temp',
        text: args.text,
        completed: false,
      },
    )
  },
})

await addTask({ text: 'Ship Trellis' })
</script>
```

## Paginated Query

```vue
<script setup lang="ts">
import { api } from '#trellis/api'

const posts = await useConvexPaginatedQuery(
  api.posts.listPaginated,
  {},
  {
    initialNumItems: 20,
    keepPreviousData: true,
  },
)
</script>

<template>
  <button
    v-if="posts.hasNextPage"
    :disabled="posts.status === 'loading-more'"
    @click="posts.loadMore(20)"
  >
    Load more
  </button>
</template>
```

## Better Auth Client Actions

```vue
<script setup lang="ts">
const { client, isAuthenticated, signOut } = useConvexAuth()

async function signInWithGitHub() {
  await client?.signIn.social({ provider: 'github' })
}
</script>
```

Use this for auth-side Better Auth actions such as OAuth provider login. Keep tenant membership, roles, and other app authorization logic in your Convex domain model instead of trying to drive it from raw auth client calls.

## Upload Queue

```vue
<script setup lang="ts">
import { api } from '#trellis/api'

const upload = useConvexUpload(api.files.generateUploadUrl, {
  maxConcurrent: 2,
  continueOnError: false,
})

async function onFilesSelected(files: FileList) {
  await upload.upload(Array.from(files))
}
</script>
```

## Nitro Server Route

```ts
import { serverConvexQuery } from '#imports'
import { api } from '#trellis/api'

export default defineEventHandler(async () => {
  return await serverConvexQuery(api.tasks.list, {})
})
```

## Convex Backend Guard

```ts
import { v } from 'convex/values'

export const remove = app.mutation({
  args: { id: v.id('posts') },
  guard: can('post.delete'),
  load: (ctx, args) => loadTenantResource(ctx.actor, ctx.db, 'posts', args.id),
  handler: async (ctx, _args, post) => {
    await ctx.db.delete(post._id)
  },
})
```
