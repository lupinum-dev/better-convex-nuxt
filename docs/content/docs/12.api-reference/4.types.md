## Status Types

### QueryStatus

Returned by `useConvexQuery().status` and `useCachedQuery().status`.

```ts
type QueryStatus = 'pending' | 'success' | 'error' | 'skipped'
```

### PaginatedQueryStatus

Returned by `useConvexPaginatedQuery().status`.

```ts
type PaginatedQueryStatus =
  | 'skipped'
  | 'loading-first-page'
  | 'ready'
  | 'loading-more'
  | 'exhausted'
  | 'error'
```

### MutationStatus

Returned by `useConvexMutation().status` and `useConvexAction().status`.

```ts
type MutationStatus = 'idle' | 'pending' | 'success' | 'error'
```

### UploadStatus

Returned by `useConvexUpload().status`.

```ts
type UploadStatus = 'idle' | 'pending' | 'success' | 'error'
```

### UploadQueueItemStatus

Status of individual items in `useConvexUpload().items`.

```ts
type UploadQueueItemStatus = 'queued' | 'pending' | 'success' | 'error' | 'cancelled'
```

## Optimistic Update Types

### OptimisticContext

Passed to `useConvexMutation({ optimisticUpdate })` callback. See guide: [Optimistic Updates](/docs/mutations/optimistic-updates).

```ts
interface OptimisticContext {
  query<Q extends FunctionReference<'query'>>(
    query: Q,
    args: FunctionArgs<Q>,
  ): OptimisticQueryHandle<Q>

  paginatedQuery<Q extends PaginatedQueryReference>(
    query: Q,
    args: PaginatedQueryArgs<Q>,
  ): OptimisticPaginatedHandle<Q>

  matchQuery<Q extends FunctionReference<'query'>>(query: Q): OptimisticMatchedQueryHandle<Q>

  matchPaginatedQuery<Q extends PaginatedQueryReference>(
    query: Q,
  ): OptimisticMatchedPaginatedHandle<Q>

  store: OptimisticLocalStore
}
```

### OptimisticQueryHandle

```ts
interface OptimisticQueryHandle<Query> {
  update: (updater: (current: Result | undefined) => Result) => void
  set: (value: Result) => void
}
```

### OptimisticPaginatedHandle

```ts
interface OptimisticPaginatedHandle<Query> {
  insertAtTop: (item: Item) => void
  insertAtPosition: (item: Item, position: number) => void
  insertAtBottomIfLoaded: (item: Item) => void
  updateItem: (id: string, updater: (item: Item) => Item) => void
  deleteItem: (id: string) => void
}
```

## Error Types

### ConvexCallError

Thrown by `useConvexMutation()` and `useConvexAction()` on failure. Also available in `trellis:mutation:error` and `trellis:action:error` hook payloads. See guide: [Error Handling](/docs/mutations/error-handling).

```ts
class ConvexCallError extends Error {
  code?: string // e.g., "LIMIT_CALLS", "UNAUTHENTICATED"
  status?: number // HTTP status code
  helper?: string // Function name
  operation?: string // "query", "mutation", "action"
  functionPath?: string // e.g., "tasks:list"
  convexUrl?: string
  authMode?: string
}
```
