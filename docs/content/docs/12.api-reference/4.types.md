## Optimistic Update Types

### OptimisticContext

```ts
interface OptimisticContext {
  query<Q extends FunctionReference<'query'>>(
    query: Q,
    args: FunctionArgs<Q>
  ): OptimisticQueryHandle<Q>

  paginatedQuery<Q extends PaginatedQueryReference>(
    query: Q,
    args: PaginatedQueryArgs<Q>
  ): OptimisticPaginatedHandle<Q>

  matchQuery<Q extends FunctionReference<'query'>>(
    query: Q
  ): OptimisticMatchedQueryHandle<Q>

  matchPaginatedQuery<Q extends PaginatedQueryReference>(
    query: Q
  ): OptimisticMatchedPaginatedHandle<Q>

  store: OptimisticLocalStore
}
```

### OptimisticQueryHandle

```ts
interface OptimisticQueryHandle<Query> {
  update: (updater: (current: Result | undefined) => Result) => void
  set: (value: Result) => void
}
```

### OptimisticPaginatedHandle

```ts
interface OptimisticPaginatedHandle<Query> {
  insertAtTop: (item: Item) => void

  insertAtBottomIfLoaded: (item: Item) => void
  updateItem: (id: string, updater: (item: Item) => Item) => void
  deleteItem: (id: string) => void
```

## Shared Schema Types

Import shared-schema types from `better-convex-nuxt/schema`.

### ConvexSchemaMeta

```ts
type ConvexSchemaMeta = {
  description?: string
}
```

Compatibility alias for the schema-level metadata base type.

### ConvexSchemaMetaFor

```ts
type ConvexSchemaMetaFor<V extends PropertyValidators> = {
  description?: string
  fields?: { [K in keyof V]: ConvexSchemaFieldMeta }
}
```

Use `ConvexSchemaMetaFor<typeof validators>` to keep field metadata locked to the exact validator keys. If `fields` is provided, it must cover every validator key.
