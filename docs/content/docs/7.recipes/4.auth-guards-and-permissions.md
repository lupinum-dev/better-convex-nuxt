## Goal

Compose route protection with an app-owned permission pattern. This module does not ship a permissions
runtime — `can()`/role checks are a thin composable you own, built on `useConvexQuery` and a Convex query
that returns the caller's role/permission context. Backend Convex functions remain the actual enforcement
point; this pattern is UX-only.

::callout{icon="i-lucide-alert-triangle" color="amber"}
**Security rule**: Never trust the frontend. Always re-check permissions inside your Convex queries/mutations.
Frontend checks only avoid rendering controls the user cannot use.
::

## Configure Route Protection

```ts [nuxt.config.ts]
export default defineNuxtConfig({
  modules: ['better-convex-nuxt'],
  convex: {
    url: process.env.CONVEX_URL,
    auth: {
      routeProtection: {
        redirectTo: '/auth/signin',
        preserveReturnTo: true,
      },
    },
  },
})
```

## Protect the Page

```ts
definePageMeta({
  convexAuth: true,
})
```

## Build Your Own Permission Composable

Define your roles/permission map next to your Convex schema:

```ts [shared/permissions.ts]
export type Role = 'owner' | 'admin' | 'member' | 'viewer'

export const permissionMap = {
  'org.settings': { roles: ['owner'] },
  'org.members': { roles: ['owner', 'admin'] },
  'post.create': { roles: ['owner', 'admin', 'member'] },
  'post.update': { own: ['member'], any: ['owner', 'admin'] },
} as const

export type Permission = keyof typeof permissionMap

export interface PermissionContext {
  userId: string
  role: Role
}

export function checkPermission(
  ctx: PermissionContext | null,
  permission: Permission,
  resource?: { ownerId?: string },
): boolean {
  if (!ctx) return false
  const rule = permissionMap[permission] as {
    roles?: readonly Role[]
    own?: readonly Role[]
    any?: readonly Role[]
  }
  if (rule.roles) return rule.roles.includes(ctx.role)
  if (rule.any?.includes(ctx.role)) return true
  if (rule.own?.includes(ctx.role)) return resource?.ownerId === ctx.userId
  return false
}
```

Then wrap a plain `useConvexQuery` call in a composable — no package runtime involved:

```ts [app/composables/useAppCapabilities.ts]
import { api } from '#convex/api'
import { checkPermission, type Permission, type PermissionContext } from '~/shared/permissions'

export function useAppCapabilities() {
  const { data: context, status } = useConvexQuery(api.auth.getPermissionContext, {})

  const can = (permission: Permission, resource?: { ownerId?: string }) =>
    checkPermission((context.value ?? null) as PermissionContext | null, permission, resource)

  const role = computed(() => context.value?.role ?? null)
  const pending = computed(() => status.value === 'pending')

  return { can, role, pending }
}
```

## Gate UI by Permission

```vue
<script setup lang="ts">
const { can, pending } = useAppCapabilities()
const canEdit = computed(() => can('post.update'))
</script>

<template>
  <button v-if="!pending && canEdit">Edit Post</button>
</template>
```

## Route Middleware Permission Check

Route middleware runs outside component setup scope, so use one-shot APIs instead of `useConvexQuery`:

- client navigation: `useConvex()`
- SSR navigation: `serverConvex(event)`

```ts [app/middleware/admin-only.ts]
import { api } from '#convex/api'

export default defineNuxtRouteMiddleware(async () => {
  const context = import.meta.server
    ? await serverConvex(useRequestEvent()!).query(api.auth.getPermissionContext, {})
    : await useConvex().query(api.auth.getPermissionContext, {})

  if (!context || (context.role !== 'admin' && context.role !== 'owner')) {
    return navigateTo('/')
  }
})
```

## Conditional Queries

Skip a query until the caller has permission:

```vue
<script setup lang="ts">
const { can } = useAppCapabilities()
const canListMembers = computed(() => can('org.members'))

const { data: members, status } = useConvexQuery(
  api.organizations.listMembers,
  computed(() => (canListMembers.value ? {} : 'skip')),
)
</script>
```
