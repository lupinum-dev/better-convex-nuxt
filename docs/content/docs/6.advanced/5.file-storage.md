## Composables

This module provides composables for file storage:

| Composable             | Description                                                      |
| ---------------------- | ---------------------------------------------------------------- |
| `useConvexFileUpload`  | Upload files with progress tracking and cancel support           |
| `useConvexUploadQueue` | Multi-file queue with concurrency control and aggregate progress |
| `useConvexStorageUrl`  | Get reactive URLs for stored files                               |

## Backend Setup

Anyone who can call your Convex functions directly (not just through your UI) can mint upload URLs, read arbitrary files, or delete arbitrary files unless your backend says otherwise. Require authentication, and track a file's owner so `getUrl`/`deleteFile` can enforce it.

Add a table to record who uploaded each file:

```ts [convex/schema.ts]
import { defineSchema, defineTable } from 'convex/server'
import { v } from 'convex/values'

export default defineSchema({
  files: defineTable({
    storageId: v.id('_storage'),
    ownerId: v.string(), // Better Auth user ID
    createdAt: v.number(),
  }).index('by_storage', ['storageId']),

  // ...your other tables
})
```

Create the required Convex functions:

```ts [convex/files.ts]
import { mutation, query } from './_generated/server'
import { v } from 'convex/values'

// Generate a short-lived upload URL - requires authentication.
export const generateUploadUrl = mutation({
  args: {},
  handler: async (ctx) => {
    const identity = await ctx.auth.getUserIdentity()
    if (!identity) {
      throw new Error('Not authenticated')
    }

    return await ctx.storage.generateUploadUrl()
  },
})

// Record ownership after upload. Call this with the storageId returned by
// `upload()` before using getUrl/deleteFile for that file.
export const saveFile = mutation({
  args: { storageId: v.id('_storage') },
  handler: async (ctx, args) => {
    const identity = await ctx.auth.getUserIdentity()
    if (!identity) {
      throw new Error('Not authenticated')
    }

    // First registration wins - nobody can contest ownership of an
    // already-registered file.
    const existing = await ctx.db
      .query('files')
      .withIndex('by_storage', (q) => q.eq('storageId', args.storageId))
      .unique()
    if (existing) {
      throw new Error('File already registered')
    }

    await ctx.db.insert('files', {
      storageId: args.storageId,
      ownerId: identity.subject,
      createdAt: Date.now(),
    })
  },
})

// Get file URL for display - only the owner can resolve their file's URL.
export const getUrl = query({
  args: { storageId: v.id('_storage') },
  handler: async (ctx, args) => {
    const identity = await ctx.auth.getUserIdentity()
    if (!identity) return null

    const file = await ctx.db
      .query('files')
      .withIndex('by_storage', (q) => q.eq('storageId', args.storageId))
      .unique()

    if (!file || file.ownerId !== identity.subject) return null

    return await ctx.storage.getUrl(args.storageId)
  },
})

// Delete a file from storage - only the owner can delete it.
export const deleteFile = mutation({
  args: { storageId: v.id('_storage') },
  handler: async (ctx, args) => {
    const identity = await ctx.auth.getUserIdentity()
    if (!identity) {
      throw new Error('Not authenticated')
    }

    const file = await ctx.db
      .query('files')
      .withIndex('by_storage', (q) => q.eq('storageId', args.storageId))
      .unique()

    if (!file || file.ownerId !== identity.subject) {
      throw new Error('Not authorized')
    }

    await ctx.db.delete(file._id)
    await ctx.storage.delete(args.storageId)
  },
})
```

::callout{icon="i-lucide-shield-alert"}
**Client validation is UX only.** `maxSize` / `allowedTypes` run in the browser and are trivially bypassed by calling your mutation directly. Enforce authentication, ownership, size, and MIME type inside your Convex functions - the upload URL and every `storageId` are attacker-reachable inputs.
::

::callout{icon="i-lucide-key-round"}
**Storage IDs are capabilities.** Treat the `storageId` returned from upload as
an unguessable capability until it is registered. In the `saveFile` example
above, the first successful registration binds that storage ID to an owner;
later callers cannot claim it because the mutation rejects existing records.
::

## useConvexFileUpload

Upload files with automatic progress tracking and cancel support.

### Return Value

```ts
const {
  upload, // (file: File, args?) => Promise<string> - returns storageId
  data, // Ref<string | undefined> - last successful storageId
  status, // ComputedRef<'idle' | 'pending' | 'success' | 'error'>
  pending, // ComputedRef<boolean>
  progress, // Ref<number> - 0 to 100
  error, // Ref<Error | null>
  cancel, // () => void - abort upload and reset state
} = useConvexFileUpload(api.files.generateUploadUrl)
```

### Options

```ts
useConvexFileUpload(mutation, {
  // Callbacks
  onSuccess: (storageId, file) => {},
  onError: (error, file) => {},
  onProgress: ({ loaded, total, percent }, file) => {},

  // Validation (checked before upload starts)
  maxSize: 5 * 1024 * 1024, // 5MB max
  allowedTypes: ['image/jpeg', 'image/png'], // Exact MIME types
  // Or use wildcards:
  // allowedTypes: ['image/*'],                    // Any image type
  // allowedTypes: ['image/*', 'application/pdf'], // Any image or PDF
})
```

### Basic Usage

```vue [components/FileUpload.vue]
<script setup lang="ts">
import { api } from '#convex/api'

const {
  upload,
  pending,
  progress,
  error,
  data: storageId,
} = useConvexFileUpload(api.files.generateUploadUrl)
const saveFile = useConvexMutation(api.files.saveFile)

async function handleFileChange(event: Event) {
  const input = event.target as HTMLInputElement
  const file = input.files?.[0]
  if (!file) return

  try {
    const id = await upload(file)
    // Record ownership so getUrl/deleteFile can authorize this file.
    await saveFile({ storageId: id })
    console.log('Uploaded:', id)
  } catch {
    // Error is automatically tracked in `error` ref
  }

  input.value = ''
}
</script>

<template>
  <div>
    <input type="file" :disabled="pending" @change="handleFileChange" />

    <div v-if="pending">Uploading: {{ progress }}%</div>

    <p v-if="error" class="error">
      {{ error.message }}
    </p>

    <p v-if="storageId">Uploaded: {{ storageId }}</p>
  </div>
</template>
```

### With Cancel Support

### With Callbacks

```vue

  onSuccess: (storageId, file) => {

```

### With Mutation Args

Pass arguments to the `generateUploadUrl` mutation (for auth, metadata, etc.):

```ts [convex/files.ts]
// Backend mutation that accepts args - still requires authentication.
export const generateUploadUrl = mutation({
  args: { type: v.string() },
  handler: async (ctx, args) => {
    const identity = await ctx.auth.getUserIdentity()
    if (!identity) {
      throw new Error('Not authenticated')
    }

    // Can use args.type for validation, logging, etc.
    return await ctx.storage.generateUploadUrl()
  },
})
```

```vue
<script setup lang="ts">
import { api } from '#convex/api'

const { upload } = useConvexFileUpload(api.files.generateUploadUrl)

async function handleAvatarUpload(file: File) {
  // Pass args to the mutation
  await upload(file, { type: 'avatar' })
}

async function handleDocumentUpload(file: File) {
  await upload(file, { type: 'document' })
}
</script>
```

### Saving to a Document

After upload, save the storageId to your data model. If your `posts` mutation already stamps an `ownerId` on the created document, that document _is_ your ownership record - you don't need the separate `files` table/`saveFile` call from [Backend Setup](#backend-setup) for files that are always attached to an owned document.

```vue
<script setup lang="ts">
import { api } from '#convex/api'

const { upload, pending, progress } = useConvexFileUpload(api.files.generateUploadUrl)
const createPost = useConvexMutation(api.posts.create)

async function handleSubmit(title: string, file: File) {
  // Upload file first
  const storageId = await upload(file)

  // Then save document with storageId (createPost stamps ownerId - see Permissions)
  await createPost({
    title,
    imageId: storageId,
  })
}
</script>
```

## useConvexUploadQueue

Queue and process many files with bounded concurrency and a single aggregate progress bar.

### Return Value

```ts
const {
  items, // Ref<UploadQueueItem[]>
  isRunning, // ComputedRef<boolean>
  hasErrors, // ComputedRef<boolean>
  queuedCount, // ComputedRef<number>
  pendingCount, // ComputedRef<number>
  successCount, // ComputedRef<number>
  errorCount, // ComputedRef<number>
  cancelledCount, // ComputedRef<number>
  aggregateProgress, // ComputedRef<number> (byte-weighted, 0-100)
  enqueue, // (input, mutationArgs?) => Promise<string[]>
  enqueueSafe, // (input, mutationArgs?) => Promise<CallResult<string[]>>
  cancelItem, // (id) => void
  cancelAll, // () => void
  clearFinished, // () => void
  reset, // () => void
} = useConvexUploadQueue(api.files.generateUploadUrl)
```

### Options

```ts
useConvexUploadQueue(api.files.generateUploadUrl, {
  maxConcurrent: 4, // Per-instance override
  continueOnError: true, // Continue scheduling after failures
  onItemSuccess: (item) => {},
  onItemError: (item) => {},
  onQueueIdle: (items) => {}, // Called when queue drains
})
```

### Enqueue Input Forms

- `File`
- `File[]`
- `FileList`
- `{ file, mutationArgs? }[]`

### Basic Usage

```vue
<script setup lang="ts">
import { api } from '#convex/api'

const { items, aggregateProgress, pendingCount, enqueue, enqueueSafe, cancelItem, clearFinished } =
  useConvexUploadQueue(api.files.generateUploadUrl, {
    maxConcurrent: 3,
  })

async function handleChange(event: Event) {
  const files = (event.target as HTMLInputElement).files
  if (!files) return

  // Programmatic flow: await all uploaded storage IDs
  const result = await enqueueSafe(files, { type: 'gallery' })
  if (!result.ok) {
    console.error(result.error.message)
    return
  }

  console.log('Uploaded storage IDs:', result.data)
}
</script>

<template>
  <div>
    <input type="file" multiple @change="handleChange" />

    <p v-if="pendingCount > 0">Uploading {{ pendingCount }} file(s): {{ aggregateProgress }}%</p>

    <ul>
      <li v-for="item in items" :key="item.id">
        {{ item.file.name }} - {{ item.status }} - {{ item.progress }}%
        <button
          v-if="item.status === 'pending' || item.status === 'queued'"
          @click="cancelItem(item.id)"
        >
          Cancel
        </button>
      </li>
    </ul>

    <button @click="clearFinished">Clear Finished</button>
  </div>
</template>
```

### Behavior Notes

- Progress is byte-weighted across all queued/pending/finished items.
- `continueOnError` defaults to `true`.
- With `continueOnError: false`, the queue stops scheduling new items after the first failure.
- `enqueue()` throws when any item fails; use `enqueueSafe()` for non-throwing programmatic flows.

## useConvexStorageUrl

Get a reactive URL for a file in storage. Automatically skips the query when storageId is null/undefined.

### API Reference

#### Parameters

| Parameter     | Type                                            | Description                                                                                                                                          |
| ------------- | ----------------------------------------------- | ---------------------------------------------------------------------------------------------------------------------------------------------------- |
| `getUrlQuery` | `FunctionReference<'query'>`                    | Convex query that accepts `{ storageId }` and returns a URL                                                                                          |
| `storageId`   | `MaybeRef<string \| null \| undefined>`         | Storage ID (reactive). Query is skipped when null/undefined.                                                                                         |
| `options`     | `{ auth?: 'required' \| 'optional' \| 'none' }` | Auth transport for the query. Use `'optional'` or `'required'` when `getUrl` requires an authenticated user (default `'none'`, behavior-preserving). |

#### Returns

| Type                          | Description                                                                       |
| ----------------------------- | --------------------------------------------------------------------------------- |
| `ComputedRef<string \| null>` | Reactive URL string, or `null` when storageId is missing or query hasn't resolved |

### Basic Usage

```vue
<script setup lang="ts">
import { api } from '#convex/api'

const storageId = ref<string | null>(null)

// URL automatically updates when storageId changes.
// api.files.getUrl requires an authenticated owner (see Backend Setup) -
// attach the caller's token with `auth: 'optional'`.
const imageUrl = useConvexStorageUrl(api.files.getUrl, storageId, { auth: 'optional' })
</script>

<template>
  <img v-if="imageUrl" :src="imageUrl" alt="Uploaded file" />
</template>
```

### With useConvexFileUpload

```vue
<script setup lang="ts">
import { api } from '#convex/api'

const {
  upload,
  pending,
  progress,
  data: storageId,
} = useConvexFileUpload(api.files.generateUploadUrl)
const saveFile = useConvexMutation(api.files.saveFile)

// URL updates automatically after successful upload
const imageUrl = useConvexStorageUrl(api.files.getUrl, storageId, { auth: 'optional' })

async function handleFile(event: Event) {
  const file = (event.target as HTMLInputElement).files?.[0]
  if (!file) return

  const id = await upload(file)
  await saveFile({ storageId: id })
}
</script>

<template>
  <input type="file" @change="handleFile" :disabled="pending" />

  <div v-if="pending">Uploading: {{ progress }}%</div>

  <img v-if="imageUrl" :src="imageUrl" />
</template>
```

### From Document Data

When a file is always attached to an owned document (see [Saving to a Document](#saving-to-a-document)), authorize `getUrl` against that document's ownership instead of the `files` table - `api.posts.get` already enforces who can read the post, so a `getUrl` that trusts "the caller could read the owning post" doesn't need a separate ownership record for this storageId.

```vue
<script setup lang="ts">
import { api } from '#convex/api'

const props = defineProps<{ postId: string }>()

// Fetch document that contains a storageId (access already checked by posts.get)
const { data: post } = await useConvexQuery(
  api.posts.get,
  computed(() => ({ id: props.postId })),
)

// Get URL from document's storageId
const imageUrl = useConvexStorageUrl(
  api.files.getUrl,
  computed(() => post.value?.imageId),
  { auth: 'optional' },
)
</script>

<template>
  <article v-if="post">
    <h1>{{ post.title }}</h1>
    <img v-if="imageUrl" :src="imageUrl" :alt="post.title" />
  </article>
</template>
```

## Complete Example

A full file upload component with progress, cancel, validation, and display:

```vue [components/ImageUploader.vue]
<script setup lang="ts">
import { api } from '#convex/api'

const {
  upload,
  pending,
  progress,
  status,
  error,
  data: storageId,
  cancel,
} = useConvexFileUpload(api.files.generateUploadUrl, {
  // Built-in validation (UX only - the backend re-checks auth/ownership)
  maxSize: 10 * 1024 * 1024, // 10MB
  allowedTypes: ['image/jpeg', 'image/png', 'image/gif', 'image/webp'],
})
const saveFile = useConvexMutation(api.files.saveFile)

const imageUrl = useConvexStorageUrl(api.files.getUrl, storageId, { auth: 'optional' })

async function handleFileChange(event: Event) {
  const input = event.target as HTMLInputElement
  const file = input.files?.[0]
  if (!file) return

  try {
    const id = await upload(file)
    await saveFile({ storageId: id })
  } catch {
    // Error tracked automatically in error ref
  }

  input.value = ''
}
</script>

<template>
  <div class="uploader">
    <!-- Upload area -->
    <div class="upload-area" :class="{ uploading: pending }">
      <input
        id="file"
        type="file"
        accept="image/*"
        :disabled="pending"
        @change="handleFileChange"
      />
      <label for="file">
        {{ pending ? 'Uploading...' : 'Choose an image' }}
      </label>

      <!-- Progress -->
      <div v-if="pending" class="progress">
        <div class="bar">
          <div class="fill" :style="{ width: `${progress}%` }" />
        </div>
        <span>{{ progress }}%</span>
        <button type="button" @click="cancel">Cancel</button>
      </div>
    </div>

    <!-- Status -->
    <p class="status">
      Status: <code>{{ status }}</code>
    </p>

    <!-- Error -->
    <p v-if="error" class="error">
      {{ error.message }}
    </p>

    <!-- Preview -->
    <div v-if="imageUrl" class="preview">
      <img :src="imageUrl" alt="Uploaded" />
      <code>{{ storageId }}</code>
    </div>
  </div>
</template>
```

## File Validation

Use built-in validation to reject files before upload starts:

```vue
<script setup lang="ts">
import { api } from '#convex/api'

const { upload, error } = useConvexFileUpload(api.files.generateUploadUrl, {
  maxSize: 10 * 1024 * 1024, // 10MB
  allowedTypes: ['image/*'], // Any image type (wildcard)
  onError: (err) => {
    // Show toast, etc.
    console.error(err.message)
  },
})
</script>

<template>
  <input type="file" @change="handleFile" />
  <p v-if="error" class="error">{{ error.message }}</p>
</template>
```

**Wildcard patterns:**

- `image/*` - Any image (jpeg, png, gif, webp, etc.)
- `video/*` - Any video
- `audio/*` - Any audio
- `application/*` - Any application type

You can mix wildcards with exact types: `['image/*', 'application/pdf']`

Validation errors are tracked in the `error` ref just like upload errors.

## Delete Files

`api.files.deleteFile` (see [Backend Setup](#backend-setup)) rejects the call unless the signed-in caller owns the file - the client doesn't need to re-check ownership before calling it.

```vue
<script setup lang="ts">
import { api } from '#convex/api'

const deleteFile = useConvexMutation(api.files.deleteFile)
const { pending } = deleteFile

async function handleDelete(storageId: string) {
  if (confirm('Delete this file?')) {
    await deleteFile({ storageId })
  }
}
</script>

<template>
  <button @click="handleDelete(storageId)" :disabled="pending">Delete</button>
</template>
```
