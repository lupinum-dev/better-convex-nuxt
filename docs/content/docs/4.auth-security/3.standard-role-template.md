::callout{icon="i-lucide-alert-triangle" color="amber"}
**Better Auth owns roles, organizations, and membership.** This scaffold reads
them from Better Auth and enforces with `auth.api.hasPermission`. There is no
app-owned `role` column and no `organizations` table.
::

---

## What You Get

A canonical org-based model on the Better Auth Organization plugin:

- Roles: `owner`, `admin`, `member`, `viewer` (defined in Better Auth)
- Backend enforcement via `auth.api.hasPermission`
- A UX-only `checkPermission()` for the frontend
- Frontend `usePermissions()` via `createPermissions`

The `starters/team` app is the runnable reference implementation for everything
below.

## 1) Enable the Organization Plugin

```ts [convex/auth.ts]
import { createAccessControl, organization } from 'better-auth/plugins'

const accessControl = createAccessControl({
  organization: ['update'],
  member: ['create', 'update', 'delete'],
  invitation: ['create', 'cancel'],
  post: ['create', 'read', 'update', 'delete'],
})

const owner = accessControl.newRole({
  organization: ['update'],
  member: ['create', 'update', 'delete'],
  invitation: ['create', 'cancel'],
  post: ['create', 'read', 'update', 'delete'],
})
const admin = accessControl.newRole({
  organization: ['update'],
  member: ['create', 'update', 'delete'],
  invitation: ['create', 'cancel'],
  post: ['create', 'read', 'update', 'delete'],
})
const member = accessControl.newRole({
  organization: [],
  member: [],
  invitation: [],
  post: ['create', 'read', 'update'],
})
const viewer = accessControl.newRole({
  organization: [],
  member: [],
  invitation: [],
  post: ['read'],
})

// inside betterAuth({ plugins: [...] }):
organization({ ac: accessControl, roles: { owner, admin, member, viewer } })
```

## 2) UX Permission Map (Frontend)

```ts [shared/permissions.ts]
export type Role = 'owner' | 'admin' | 'member' | 'viewer'

export const permissions = {
  'org.settings': { roles: ['owner'] },
  'org.invite': { roles: ['owner', 'admin'] },
  'org.members': { roles: ['owner', 'admin'] },
  'post.create': { roles: ['owner', 'admin', 'member'] },
  'post.read': { roles: ['owner', 'admin', 'member', 'viewer'] },
  'post.update': { own: ['member'], any: ['owner', 'admin'] },
  'post.delete': { own: ['member'], any: ['owner', 'admin'] },
  'post.publish': { roles: ['owner', 'admin'] },
} as const

export type Permission = keyof typeof permissions

export interface PermissionContext {
  role: Role
  userId: string
  orgId: string
}

export interface Resource {
  ownerId?: string
}

type Rule = { roles: readonly Role[] } | { own: readonly Role[]; any: readonly Role[] }

export function checkPermission(
  ctx: PermissionContext | null,
  permission: Permission,
  resource?: Resource,
): boolean {
  if (!ctx) return false
  const rule = permissions[permission] as Rule

  if ('roles' in rule) return (rule.roles as readonly string[]).includes(ctx.role)
  if ((rule.any as readonly string[]).includes(ctx.role)) return true
  if (!resource?.ownerId) return false
  return (rule.own as readonly string[]).includes(ctx.role) && resource.ownerId === ctx.userId
}
```

## 3) Backend Authorization Helper

```ts [convex/lib/permissions.ts]
import { ConvexError } from 'convex/values'

import { components } from '../_generated/api'
import type { MutationCtx, QueryCtx } from '../_generated/server'
import { authComponent, createAuth } from '../auth'

type Ctx = QueryCtx | MutationCtx
type PostPermission = 'create' | 'read' | 'update' | 'delete'

export async function getBetterAuthMember(
  ctx: Ctx,
  args: { organizationId: string; userId: string },
) {
  return (await ctx.runQuery(components.betterAuth.adapter.findOne, {
    model: 'member',
    where: [
      { field: 'organizationId', value: args.organizationId },
      { field: 'userId', value: args.userId },
    ],
  })) as { role: string } | null
}

export async function requirePostPermission(
  ctx: Ctx,
  args: { organizationId: string; permission: PostPermission },
) {
  const { auth, headers } = await authComponent.getAuth(createAuth, ctx)
  const session = await auth.api.getSession({ headers })
  if (!session) {
    throw new ConvexError({ code: 'UNAUTHENTICATED', message: 'Not authenticated' })
  }

  const allowed = await auth.api.hasPermission({
    headers,
    body: { organizationId: args.organizationId, permissions: { post: [args.permission] } },
  })
  if (!allowed.success) {
    throw new ConvexError({
      code: 'FORBIDDEN',
      message: `Missing post:${args.permission} permission`,
    })
  }

  return { userId: session.user.id }
}
```

## 4) Permission Context Query

```ts [convex/auth.ts]
import { query } from './_generated/server'
import { getBetterAuthMember } from './lib/permissions'

export const getPermissionContext = query({
  args: {},
  handler: async (ctx) => {
    const { auth, headers } = await authComponent.getAuth(createAuth, ctx)
    const session = await auth.api.getSession({ headers })
    if (!session) return null

    const organizationId = session.session.activeOrganizationId
    if (!organizationId) return null

    const member = await getBetterAuthMember(ctx, { organizationId, userId: session.user.id })
    if (!member) return null

    return { role: member.role, userId: session.user.id, orgId: organizationId }
  },
})
```

## 5) Frontend Composable

```ts [composables/usePermissions.ts]
import { api } from '#convex/api'
import {
  checkPermission,
  type Permission,
  type PermissionContext,
  type Resource,
} from '~~/shared/permissions'

export const { usePermissions, usePermissionRedirect } = createPermissions<
  Permission,
  PermissionContext,
  Resource
>({
  query: api.auth.getPermissionContext,
  checkPermission,
})
```

Use in components:

```vue
<script setup lang="ts">
const { can, role } = usePermissions()
</script>

<template>
  <button v-if="can('post.create')">New Post</button>
  <button v-if="can('org.invite')">Invite Member</button>
</template>
```

## Next Customizations

- Add resource groups beyond `post` (for example `comment`, `invoice`, `project`)
  to both the access-control statements and the UX permission map.
- Enforce every org-scoped mutation with `requirePostPermission`-style helpers.
- Manage roles and invitations through the Better Auth API
  (`auth.api.updateMemberRole`, `auth.api.createInvitation`), never by patching
  an app table.
